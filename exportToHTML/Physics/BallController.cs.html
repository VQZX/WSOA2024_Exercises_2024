1    using System;
2    using UnityEngine;
3    using UnityEngine.EventSystems;
4    using UnityEngine.PlayerLoop;
5    
6    namespace Physics_Demonstration
7    {
8        public class BallController : MonoBehaviour, IDragHandler, IEndDragHandler
9        {
10           [SerializeField]
11           private Rigidbody2D attachedBody;
12           public Rigidbody2D AttachedBody => attachedBody;
13           
14           [SerializeField]
15           private Camera mainCamera;
16   
17           [SerializeField]
18           private float forceMultiplier = 1f;
19   
20           [SerializeField]
21           private float maxForceMagnitude = 50;
22   
23           [SerializeField]
24           private float minimumSpeedToStop = 1f;
25           
26           private Vector3 mouseToBall;
27           
28           private enum BallState
29           {
30               Nothing,
31               CanInteract,
32               Pulling,
33               Moving
34           }
35   
36           private BallState state = BallState.CanInteract;
37           
38           private float previousMagnitude;
39           private float currentMagnitude;
40   
41           
42           public void OnDrag(PointerEventData eventData)
43           {
44               if (state != BallState.CanInteract)
45               {
46                   return;
47               }
48   
49               state = BallState.Pulling;
50               Vector3 cursourWorldPosition = mainCamera.ScreenToWorldPoint(eventData.position);
51               cursourWorldPosition.z = 0;
52   
53               mouseToBall = transform.position - cursourWorldPosition;
54   
55           }
56   
57           public void OnEndDrag(PointerEventData eventData)
58           {
59               var calculatedForce = mouseToBall * forceMultiplier;
60               var magnitude = calculatedForce.magnitude;
61               if (magnitude > maxForceMagnitude)
62               {
63                   calculatedForce = mouseToBall.normalized * maxForceMagnitude;
64               }
65               attachedBody.AddForce(calculatedForce, ForceMode2D.Impulse);
66               state = BallState.Moving;
67               currentMagnitude = attachedBody.velocity.magnitude;
68           }
69   
70           private void Start()
71           {
72               if (attachedBody == null)
73               {
74                   attachedBody = GetComponent<Rigidbody2D>();
75               }
76           }
77   
78           private void Update()
79           {
80               if (state == BallState.Moving)
81               {
82                   MeasureSpeedToStop();
83               }
84               
85           }
86   
87           private void MeasureSpeedToStop()
88           {
89               if (attachedBody == null)
90               {
91                   return;
92               }
93   
94               previousMagnitude = currentMagnitude;
95               currentMagnitude = attachedBody.velocity.magnitude;
96               float acceleration = currentMagnitude - previousMagnitude;
97               Debug.Log(acceleration);
98               // If the current acceleration is negative, start checking
99               if (acceleration < 0)
100              {
101                  Debug.Log($"Current Speed: {attachedBody.velocity.magnitude}");
102                  var speed = attachedBody.velocity.magnitude;
103                  if (speed < minimumSpeedToStop)
104                  {
105                      attachedBody.velocity = Vector3.zero;
106                      state = BallState.CanInteract;
107                  }
108              }
109              
110          }
111      }
112  }
