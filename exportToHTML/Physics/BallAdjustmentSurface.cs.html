1    using System;
2    using UnityEngine;
3    using Random = UnityEngine.Random;
4    
5    namespace Physics_Demonstration
6    {
7        public class BallAdjustmentSurface : MonoBehaviour
8        {
9            private enum Behaviour
10           {
11               Nothing,
12               SpeedUp,
13               SlowDown,
14               Destroy,
15               Teleport
16           }
17   
18           [SerializeField]
19           private BallManager ballManager;
20           
21           [SerializeField]
22           private Behaviour behaviour;
23   
24           [SerializeField]
25           private float speedUpMultiplier = 2f;
26   
27           [SerializeField]
28           private float slowDownMultiplier = 2f;
29   
30           [SerializeField]
31           private float teleportRadius = 2f;
32           
33           private void OnTriggerEnter2D(Collider2D other)
34           {
35               BallController ball = other.GetComponent<BallController>();
36               if (ball == null)
37               {
38                   return;
39               }
40   
41               switch (behaviour)
42               {
43                   case Behaviour.Nothing:
44                       break;
45                   case Behaviour.SpeedUp:
46                       SpeedUp(ball);
47                       break;
48                   case Behaviour.SlowDown:
49                       SlowDown(ball);
50                       break;
51                   case Behaviour.Destroy:
52                       DestroyBall(ball);
53                       break;
54                   case Behaviour.Teleport:
55                       Telelport(ball);
56                       break;
57                   default:
58                       throw new ArgumentOutOfRangeException();
59               }
60           }
61           
62           private void SpeedUp(BallController ball)
63           {
64               var velocity = ball.AttachedBody.velocity;
65               var magnitude = velocity.magnitude;
66               var direction = velocity.normalized;
67   
68               var newVelocity = direction * magnitude * speedUpMultiplier;
69               ball.AttachedBody.velocity = newVelocity;
70           }
71           
72           private void SlowDown(BallController ball)
73           {
74               var velocity = ball.AttachedBody.velocity;
75               var magnitude = velocity.magnitude;
76               var direction = velocity.normalized;
77   
78               var newVelocity = direction * magnitude / slowDownMultiplier;
79               ball.AttachedBody.velocity = newVelocity;
80           }
81       
82           private void DestroyBall(BallController ball)
83           {
84               ballManager.RespawnBall();
85           }
86           
87           private void Telelport(BallController ball)
88           {
89               var randomCirclePoint = Random.insideUnitCircle;
90               Vector3 worldPoint = new Vector3(randomCirclePoint.x, randomCirclePoint.y, 0);
91               var vectorTo = ball.transform.position + worldPoint * teleportRadius;
92               ball.transform.position = vectorTo;
93           }
94    
95       }
96   }
