1    using System;
2    using System.Collections;
3    using System.Collections.Generic;
4    using UnityEngine;
5    using Random = UnityEngine.Random;
6    
7    public class SquareController : MonoBehaviour
8    {
9        [SerializeField]
10       private float frequencyOfColorChange;
11   
12       [SerializeField]
13       private SpriteRenderer spriteRender;
14   
15       [SerializeField]
16       private float movementSpeed = 5f;
17   
18       [SerializeField]
19       private float radiusOfMotion = 1;
20       
21       private float currentTime;
22       private Vector3 initialLocalPosition;
23   
24   
25       private void Start()
26       {
27           radiusOfMotion = Random.Range(3, 56);
28           movementSpeed = Random.Range(1, 12);
29           initialLocalPosition = transform.localPosition;
30       }
31   
32       private void Update()
33       {
34           ChangeColorByTimer();
35   
36           HandleMovement();
37       }
38   
39       private void HandleMovement()
40       {
41           float x = initialLocalPosition.x + radiusOfMotion * Mathf.Cos(Time.time * movementSpeed);
42           float y = initialLocalPosition.y + radiusOfMotion * Mathf.Sin(Time.time * movementSpeed);
43   
44           transform.localPosition = new Vector3(x, y);
45   
46       }
47   
48       private void ChangeColorByTimer()
49       {
50           currentTime += Time.deltaTime;
51           if (currentTime < frequencyOfColorChange)
52           {
53               return;
54           }
55           
56           // Random.ColorHSV(minHue, maxHue, minSaturation, maxSaturation, minValue, maxValue)
57           Color color = Random.ColorHSV(0, 1, 1, 1, 1, 1);
58           spriteRender.color = color;
59           currentTime = 0;
60       }
61   }
