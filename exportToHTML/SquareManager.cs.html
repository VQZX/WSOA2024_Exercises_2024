1    using System;
2    using System.Collections;
3    using System.Collections.Generic;
4    using UnityEngine;
5    using Random = UnityEngine.Random;
6    
7    public class SquareManager : MonoBehaviour
8    {
9        [SerializeField]
10       private SquareController template;
11   
12       [SerializeField]
13       private int amountOfClones = 10;
14   
15       [SerializeField]
16       private float positionMultiplier = 3;
17   
18       private List<SquareController> squares = new List<SquareController>();
19       private float initialX;
20   
21       private void Start()
22       {
23           initialX = transform.position.x;
24           for (int i = 0; i < amountOfClones; i++)
25           {
26               Vector2 randomPosition = Random.insideUnitCircle * positionMultiplier;
27               var position = transform.position + new Vector3(randomPosition.x, randomPosition.y, 0);
28               
29               //var clone = Instantiate(template, position, Quaternion.identity);
30               var clone = Instantiate(template, transform);
31               clone.transform.localPosition = position;
32               squares.Add(clone);
33           }
34       }
35   
36       private void Update()
37       {
38           var deltaX = 10 * Mathf.Sin(Time.time * 2);
39           var nextX = initialX + deltaX;
40           transform.position = new Vector3(nextX, transform.position.y, 0);
41       }
42   }
