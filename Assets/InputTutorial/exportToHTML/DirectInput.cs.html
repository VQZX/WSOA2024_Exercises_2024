DirectInput.cs 
1    using System;
2    using System.Collections;
3    using System.Collections.Generic;
4    using InputTutorial;
5    using Unity.VisualScripting;
6    using UnityEngine;
7    using UnityEngine.InputSystem;
8    using UnityEngine.InputSystem.Controls;
9    using Quaternion = UnityEngine.Quaternion;
10   using Vector3 = UnityEngine.Vector3;
11   
12   public class DirectInput : MonoBehaviour
13   {
14   
15       [SerializeField]
16       private MovementControl movementControl;
17       
18       [SerializeField]
19       private float mouseDeadZone = 0.05f;
20       
21       private Keyboard currentKeyboard = Keyboard.current;
22       private Mouse currentMouse = Mouse.current;
23   
24       private void Update()
25       {
26           if (CanTranslate())
27           {
28               ManageTranslation();
29           }
30   
31           if (currentMouse.delta.magnitude > mouseDeadZone)
32           {
33               ManageOrientation(currentMouse.delta);
34           }
35       }
36   
37       private bool CanTranslate()
38       {
39           var output = currentKeyboard.wKey.isPressed ||
40                        currentKeyboard.sKey.isPressed ||
41                        currentKeyboard.aKey.isPressed ||
42                        currentKeyboard.dKey.isPressed;
43           return output;
44       }
45   
46       private void ManageTranslation()
47       {
48           Vector2 inputDirection = new Vector2();
49           if (currentKeyboard.wKey.isPressed)
50           {
51               inputDirection += Vector2.up;
52           }
53           if (currentKeyboard.sKey.isPressed)
54           {
55               inputDirection += Vector2.down;
56           }
57           if (currentKeyboard.aKey.isPressed)
58           {
59               inputDirection += Vector2.left;
60           }
61           if (currentKeyboard.dKey.isPressed)
62           {
63               inputDirection += Vector2.right;
64           }
65   
66           inputDirection = inputDirection.normalized;
67           
68           
69           movementControl.ManageTranslation(inputDirection);
70       }
71       
72       private void ManageOrientation(DeltaControl mouseDelta)
73       {
74           var delta = new Vector2(mouseDelta.x.value, mouseDelta.y.value);
75           movementControl.ManageOrientation(delta);
76       }
77   }
