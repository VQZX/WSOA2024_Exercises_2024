1    using UnityEngine;
2    
3    public class BasicCameraFollow : MonoBehaviour
4    {
5        [SerializeField]
6        private Transform target;
7    
8        [SerializeField]
9        private Vector3 offset;
10       
11       private void Update()
12       {
13           if (target == null)
14           {
15               return;
16           }
17           
18           // Look at the target
19           transform.LookAt(target);
20           SelectNextPosition();
21   
22       }
23       
24       private void SelectNextPosition()
25       {
26           // Orbit around the player
27           /* 
28            * Rotation in 3D around the y-component to get the correct forward orientation 
29            * Because they y-component goes upwards in Unity, 
30            * We only have to worry about adjust the x and z values. 
31            * As such we only rotate around the y-component in 3D as such 
32            * per vector-matrix calculation 
33            * x = motionVector.x * cos(theta) + 0 + motionVector.z * sin(theta) 
34            * y = 0 + motionVector.y + 0 
35            * z = motionVector.x * (-sin(theta)) + 0 + motionVector.z * cos(theta) 
36            * 
37            * Where theta is the targets y-rotation in the euler angles 
38            */
39           var yRotation = target.localRotation.eulerAngles.y * Mathf.Deg2Rad;
40           var xPrime = offset.x * Mathf.Cos(yRotation) + offset.z * Mathf.Sin(yRotation);
41           var zPrime = -offset.x * Mathf.Sin(yRotation) + offset.z * Mathf.Cos(yRotation);
42   
43           var rotatedOffsetVector = new Vector3(xPrime, offset.y, zPrime);
44           transform.localPosition = target.position + rotatedOffsetVector;
45       }
46   }
