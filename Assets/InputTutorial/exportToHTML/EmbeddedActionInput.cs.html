1    using System;
2    using UnityEngine;
3    using UnityEngine.InputSystem;
4    
5    namespace InputTutorial
6    {
7        public class EmbeddedActionInput : MonoBehaviour
8        {
9            [SerializeField]
10           private MovementControl movementControl;
11           
12           [SerializeField]
13           private InputAction move;
14   
15           [SerializeField]
16           private InputAction turn;
17   
18           private Vector2 moveDelta;
19           
20           private void Awake()
21           {
22               move.performed += MovementAction;
23               turn.performed += TurnAction;
24           }
25   
26           private void Update()
27           {
28               if (moveDelta.sqrMagnitude > 0)
29               {
30                   movementControl.ManageTranslation(moveDelta);
31               }
32   
33               if (move.WasReleasedThisFrame())
34               {
35                   moveDelta = Vector2.zero;
36               }
37           }
38   
39           private void MovementAction(InputAction.CallbackContext context)
40           {
41               var inputData = context.ReadValue<Vector2>();
42               moveDelta = inputData;
43               movementControl.ManageTranslation(inputData);
44           }
45   
46           private void TurnAction(InputAction.CallbackContext context)
47           {
48               var inputData = context.ReadValue<Vector2>();
49               movementControl.ManageOrientation(inputData);
50           }
51   
52           private void OnEnable()
53           {
54               move.Enable();
55               turn.Enable();
56           }
57   
58           private void OnDisable()
59           {
60               move.Disable();
61               turn.Disable();
62           }
63       }
64   }
